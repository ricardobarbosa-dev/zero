💡 Regra de ouro

Cada classe deve ter uma única razão para mudar. Se a classe faz autenticação E envia emails E salva no banco — ela está errada. Quebre em classes menores.

@dataclass + ABC + @property

from dataclasses import dataclass, field
from abc import ABC, abstractmethod
from typing import ClassVar

# ABC define o contrato — não pode ser instanciada
class Forma(ABC):
    @abstractmethod
    def area(self) -> float: ...

    @abstractmethod
    def perimetro(self) -> float: ...

    def descricao(self) -> str:
        # método concreto — reutilizado por subclasses
        return f"{type(self).__name__}: área={self.area():.2f}"


# @dataclass gera __init__, __repr__, __eq__ automaticamente
@dataclass
class Retangulo(Forma):
    largura: float
    altura:  float
    _total_criados: ClassVar[int] = 0

    def __post_init__(self):
        if self.largura <= 0 or self.altura <= 0:
            raise ValueError("Dimensões devem ser positivas")
        Retangulo._total_criados += 1

    def area(self) -> float:
        return self.largura * self.altura

    def perimetro(self) -> float:
        return 2 * (self.largura + self.altura)

    @property
    def eh_quadrado(self) -> bool:
        # @property — acessa como atributo: r.eh_quadrado (sem ())
        return self.largura == self.altura

    @classmethod
    def quadrado(cls, lado: float) -> "Retangulo":
        # Factory method — forma conveniente de criar
        return cls(lado, lado)

    @staticmethod
    def validar(d: float) -> bool:
        # Sem self/cls — utilitário puro
        return d > 0


r = Retangulo(5.0, 3.0)
print(r.area())        # 15.0
print(r.eh_quadrado)   # False  (sem parênteses!)
print(r.descricao())   # Retangulo: área=15.00

q = Retangulo.quadrado(4.0)
print(q.eh_quadrado)   # True

Herança e @property com setter

class Conta:
    def __init__(self, titular: str, saldo: float = 0):
        self._titular = titular
        self._saldo   = saldo    # _ = "protegido" por convenção

    @property
    def saldo(self) -> float:
        return self._saldo

    @saldo.setter
    def saldo(self, valor: float):
        if valor < 0:
            raise ValueError("Saldo não pode ser negativo")
        self._saldo = valor

    def depositar(self, valor: float) -> None:
        if valor <= 0:
            raise ValueError("Valor deve ser positivo")
        self._saldo += valor

    def sacar(self, valor: float) -> None:
        if valor > self._saldo:
            raise ValueError("Saldo insuficiente")
        self._saldo -= valor

    def __str__(self) -> str:
        return f"Conta({self._titular}, R${self._saldo:.2f})"


# Herança — ContaEspecial herda tudo de Conta
class ContaEspecial(Conta):
    def __init__(self, titular: str, limite: float = 500):
        super().__init__(titular)   # chama __init__ do pai
        self._limite = limite

    def sacar(self, valor: float) -> None:
        # Override: permite cheque especial até o limite
        if valor > self._saldo + self._limite:
            raise ValueError("Limite excedido")
        self._saldo -= valor


c = Conta("Alice")
c.depositar(1000)
c.sacar(200)
print(c.saldo)   # 800.0
print(c)         # Conta(Alice, R$800.00)

Decorators

💡 O que é um decorator?

Uma função que recebe outra função, adiciona comportamento e retorna. Usado em Django (@login_required), FastAPI (@app.get), pytest (@pytest.fixture).

import functools, time, logging

# Decorator simples
def log_chamada(func):
    @functools.wraps(func)   # preserva __name__ e __doc__
    def wrapper(*args, **kwargs):
        print(f"→ Chamando {func.__name__}")
        resultado = func(*args, **kwargs)
        print(f"← Terminou {func.__name__}")
        return resultado
    return wrapper

@log_chamada
def somar(a: int, b: int) -> int:
    return a + b

somar(2, 3)
# → Chamando somar
# ← Terminou somar

# Decorator com parâmetros (factory de decorator)
def retry(max_tentativas: int = 3, delay: float = 1.0):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for t in range(1, max_tentativas + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    logging.warning(f"Tentativa {t}: {e}")
                    if t == max_tentativas: raise
                    time.sleep(delay)
        return wrapper
    return decorator

@retry(max_tentativas=3, delay=0.5)
def buscar_api(url: str) -> dict:
    import requests
    r = requests.get(url, timeout=5)
    r.raise_for_status()
    return r.json()

# lru_cache — memoização built-in
from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n: int) -> int:
    if n <= 1: return n
    return fibonacci(n-1) + fibonacci(n-2)

fibonacci(35)   # instantâneo — usa cache

Generators — memória O(1)

💡 Por que usar generators?

Uma lista com 1 milhão de itens ocupa ~8MB. O generator equivalente ocupa ~200 bytes — porque processa um item por vez, não carrega tudo na memória.

from itertools import islice, groupby

# yield — pausa e entrega um valor por vez
def ler_csv(caminho: str):
    import csv
    with open(caminho, encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for linha in reader:
            yield linha   # 1 linha na memória por vez

# Funciona com 10 milhões de linhas sem explodir RAM
total = 0
for row in ler_csv("vendas.csv"):
    total += float(row["valor"])

# Generator expression (mais leve que list comprehension)
soma = sum(x**2 for x in range(1_000_000))

# islice — pegar N itens de iterável (até infinito)
def contar(n: int = 0):
    while True:
        yield n
        n += 1

primeiros_5 = list(islice(contar(10), 5))
# [10, 11, 12, 13, 14]

# groupby — agrupar por chave (precisa estar ordenado!)
pedidos = [
    {"status": "pago",     "id": 1},
    {"status": "pago",     "id": 3},
    {"status": "pendente", "id": 2},
]
pedidos.sort(key=lambda x: x["status"])
for status, grupo in groupby(pedidos, key=lambda x: x["status"]):
    ids = [p["id"] for p in grupo]
    print(f"{status}: {ids}")
# pago: [1, 3]
# pendente: [2]

Context Managers

from contextlib import contextmanager
import time

@contextmanager
def timer(label: str = "Execução"):
    inicio = time.perf_counter()
    try:
        yield    # código dentro do "with" roda aqui
    finally:
        elapsed = time.perf_counter() - inicio
        print(f"{label}: {elapsed*1000:.1f}ms")
        # finally garante que imprime mesmo se der erro

with timer("Busca no banco"):
    resultado = buscar_produtos()
# Busca no banco: 45.3ms

# Context manager para transação
@contextmanager
def transacao(conn):
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise

with transacao(db_conn) as conn:
    conn.execute("UPDATE contas SET saldo = saldo - 100 WHERE id = 1")
    conn.execute("UPDATE contas SET saldo = saldo + 100 WHERE id = 2")
# commit se ok, rollback automático se erro

Async / Await

Execute N tarefas simultâneas sem criar threads — ideal para I/O intenso.

💡 Quando usar async?

I/O-bound: chamadas HTTP, banco de dados, arquivos. Sem async: 10 chamadas × 500ms = 5s. Com async: 10 chamadas simultâneas ≈ 500ms total.

import asyncio
import aiohttp

# async def — define uma coroutine
async def buscar_usuario(session: aiohttp.ClientSession, uid: int) -> dict:
    url = f"https://jsonplaceholder.typicode.com/users/{uid}"
    async with session.get(url) as resp:

        return [r for r in resultados if not isinstance(r, Exception)]

# asyncio.Queue — produtor / consumidor desacoplados
async def produtor(queue: asyncio.Queue, itens: list) -> None:
    for item in itens:
        await queue.put(item)
    await queue.put(None)   # sentinel = sinal de fim

async def consumidor(queue: asyncio.Queue) -> None:
    while True:
        item = await queue.get()
        if item is None:
            break
        print(f"Processando: {item}")
        await asyncio.sleep(0.1)   # simula I/O
        queue.task_done()

async def main() -> None:
    queue: asyncio.Queue = asyncio.Queue(maxsize=5)
    await asyncio.gather(
        produtor(queue, [1, 2, 3, 4, 5]),
        consumidor(queue),
    )

asyncio.run(main())

import time

def rate_limit(user_id: str, max_req: int = 10, window: int = 60) -> bool:
    key = f"rl:{user_id}"
    now = time.time()
    pipe = r.pipeline()
    pipe.zremrangebyscore(key, 0, now - window)   # remove expirados
    pipe.zadd(key, {str(now): now})
    pipe.expire(key, window)
    pipe.zcard(key)
    results = pipe.execute()
    return results[-1] <= max_req

JOINs e CTEs

-- INNER JOIN — só linhas com correspondência nos dois lados
SELECT p.nome, c.nome AS categoria
FROM produtos p
INNER JOIN categorias c ON p.categoria_id = c.id
WHERE p.preco > 100;

-- LEFT JOIN — todos da esquerda, null se não tiver match
SELECT u.email, COUNT(p.id) AS total_pedidos
FROM usuarios u
LEFT JOIN pedidos p ON p.usuario_id = u.id
GROUP BY u.email
ORDER BY total_pedidos DESC;

-- CTE (WITH) — subquery nomeada, mais legível que subquery
WITH pedidos_recentes AS (
    SELECT usuario_id, SUM(total) AS total_gasto
    FROM pedidos
    WHERE criado_em >= NOW() - INTERVAL '30 days'
    GROUP BY usuario_id
),
top_clientes AS (
    SELECT usuario_id FROM pedidos_recentes
    WHERE total_gasto > 500
    ORDER BY total_gasto DESC
    LIMIT 10
)
SELECT u.email, pr.total_gasto
FROM usuarios u
JOIN pedidos_recentes pr ON pr.usuario_id = u.id
WHERE u.id IN (SELECT usuario_id FROM top_clientes);

Window Functions

-- ROW_NUMBER — top N por categoria
SELECT nome, categoria_id, preco,
       ROW_NUMBER() OVER (PARTITION BY categoria_id ORDER BY preco DESC) AS rank_na_cat
FROM produtos;

-- Filtrar: top 3 de cada categoria
SELECT * FROM (
    SELECT nome, categoria_id, preco,
           ROW_NUMBER() OVER (PARTITION BY categoria_id ORDER BY preco DESC) AS rn
    FROM produtos
) t WHERE rn <= 3;

-- LAG — comparar com a linha anterior
SELECT mes, receita,
       LAG(receita) OVER (ORDER BY mes) AS mes_anterior,
       ROUND((receita - LAG(receita) OVER (ORDER BY mes)) / LAG(receita) OVER (ORDER BY mes) * 100, 1) AS variacao_pct
FROM receitas_mensais;

-- SUM cumulativo (running total)
SELECT data, valor,
       SUM(valor) OVER (ORDER BY data ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS acumulado
FROM vendas;

async def listar_produtos(db: AsyncSession, categoria_id: int | None = None):
    stmt = select(Produto).order_by(Produto.nome)
    if categoria_id:
        stmt = stmt.where(Produto.categoria_id == categoria_id)
    result = await db.execute(stmt)

async def criar_produto(db: AsyncSession, dados: dict) -> Produto:
    produto = Produto(**dados)
    db.add(produto)
    await db.commit()
    await db.refresh(produto)
    return produto

async def atualizar_produto(db: AsyncSession, produto_id: int, dados: dict) -> Produto | None:
    produto = await db.get(Produto, produto_id)
    if not produto: return None
    for k, v in dados.items():
        setattr(produto, k, v)
    await db.commit()
    await db.refresh(produto)
    return produto

from abc import ABC, abstractmethod

class StorageBackend(ABC):
    @abstractmethod
    def salvar(self, nome: str, conteudo: bytes) -> str: ...

class LocalStorage(StorageBackend):
    def salvar(self, nome: str, conteudo: bytes) -> str:
        path = f"/media/{nome}"
        with open(path, "wb") as f: f.write(conteudo)
        return path

class S3Storage(StorageBackend):
    def salvar(self, nome: str, conteudo: bytes) -> str:
        s3 = boto3.client("s3")
        s3.put_object(Bucket="my-bucket", Key=nome, Body=conteudo)
        return f"https://my-bucket.s3.amazonaws.com/{nome}"

# Factory com registro — extensível sem modificar a factory
class StorageFactory:
    _backends: dict = {}

    @classmethod
    def registrar(cls, nome: str, backend: type) -> None:
        cls._backends[nome] = backend

    @classmethod
    def criar(cls, nome: str) -> StorageBackend:
        if nome not in cls._backends:
            raise ValueError(f"Backend {nome!r} não registrado")
        return cls._backends[nome]()

StorageFactory.registrar("local", LocalStorage)
StorageFactory.registrar("s3",    S3Storage)

# Uso — sem saber qual classe vai criar
storage = StorageFactory.criar(settings.STORAGE_BACKEND)
url = storage.salvar("foto.jpg", conteudo)

Strategy — algoritmos intercambiáveis

from typing import Protocol

class RegraDesconto(Protocol):
    def calcular(self, total: float) -> float: ...

class SemDesconto:
    def calcular(self, total: float) -> float:
        return total

class DescontoPorcentagem:
    def __init__(self, pct: float): self.pct = pct
    def calcular(self, total: float) -> float:
        return total * (1 - self.pct / 100)

class DescontoValorFixo:
    def __init__(self, valor: float): self.valor = valor
    def calcular(self, total: float) -> float:
        return max(0, total - self.valor)

class Carrinho:
    def __init__(self, regra: RegraDesconto = SemDesconto()):
        self._regra = regra
        self._itens: list[float] = []

    def adicionar(self, preco: float) -> None:
        self._itens.append(preco)

    def total(self) -> float:
        return self._regra.calcular(sum(self._itens))

# Trocar estratégia sem mudar o Carrinho
c = Carrinho(DescontoPorcentagem(10))
c.adicionar(100)
c.adicionar(200)
print(c.total())   # 270.0  (10% off)
